const canvas = document.querySelector('#shader-canvas');
const gl = canvas.getContext('webgl2');

/* Fragment shader: 6-layer FBM noise, aurora palette */
float fbm(vec2 p, float t) {
  float val = 0.0; float amp = 0.5; float freq = 1.0;
  for (int i = 0; i < 5; i++) {
    val += amp * snoise(p * freq + t * 0.3);
    freq *= 2.0; amp *= 0.5;
  }
  return val;
}

  gl.drawArrays(gl.TRIANGLES, 0, 6);
  requestAnimationFrame(render);
}

let currentY = 0, targetY = 0;
const LERP = 0.08; // 0.05 = silky, 0.2 = responsive

container.addEventListener('wheel', e => {
  e.preventDefault();
  targetY = Math.max(0, Math.min(maxScroll, targetY + e.deltaY));
}, { passive: false });

function tick() {
  currentY += (targetY - currentY) * LERP;
  inner.style.transform = `translateY(-${currentY}px)`;
  requestAnimationFrame(tick);
}
